CREATE PROCEDURE sp{TABLE.NAME}_Select(
{TABLE.COLUMNS PRIMARY}
@{COLUMN.NAME}  {COLUMN.TYPE} {/TABLE.COLUMNS}
)
AS
SELECT {TABLE.COLUMNS}{COLUMN.NAME}{IF NOT LAST}, {/IF}{/TABLE.COLUMNS}
WHERE {TABLE.COLUMNS PRIMARY}{COLUMN.NAME}=@{COLUMN.NAME}{/TABLE.COLUMNS}
GO


CREATE PROCEDURE sp{TABLE.NAME}_Insert (  
{TABLE.COLUMNS NOPRIMARY}
@{COLUMN.NAME}  {COLUMN.TYPE} {IF COLUMN.TYPE EQ 'varchar|nvarchar'}({COLUMN.LENGTH}){/IF},{/TABLE.COLUMNS}
{TABLE.COLUMNS PRIMARY}
@{COLUMN.NAME}  {COLUMN.TYPE} {IF COLUMN.TYPE EQ 'varchar|nvarchar'}({COLUMN.LENGTH}){/IF} OUTPUT  {/TABLE.COLUMNS}
)   
AS  
 INSERT INTO {TABLE.NAME} ({TABLE.COLUMNS NOPRIMARY}{COLUMN.NAME}{IF NOT LAST},{/IF}{/TABLE.COLUMNS})  
 VALUES ({TABLE.COLUMNS NOPRIMARY}@{COLUMN.NAME}{IF NOT LAST},{/IF}{/TABLE.COLUMNS})  
 {TABLE.COLUMNS PRIMARY}
 SET @{COLUMN.NAME} = SCOPE_IDENTITY()  {/TABLE.COLUMNS}
GO



CREATE PROCEDURE sp{TABLE.NAME}_Update  
{TABLE.COLUMNS}@{COLUMN.NAME} {COLUMN.TYPE}{IF COLUMN.TYPE EQ 'varchar|nvarchar'}({COLUMN.LENGTH}){/IF}{IF NOT LAST},{/IF}
{/TABLE.COLUMNS}
AS  
  
 UPDATE {TABLE.NAME} SET {TABLE.COLUMNS NOPRIMARY}{COLUMN.NAME} = @{COLUMN.NAME}{IF NOT LAST},{/IF}{/TABLE.COLUMNS}
 WHERE {TABLE.COLUMNS PRIMARY}{COLUMN.NAME}= @{COLUMN.NAME}{/TABLE.COLUMNS}
GO


CREATE PROCEDURE sp{TABLE.NAME}_Delete (  
{TABLE.COLUMNS PRIMARY}@{COLUMN.NAME} {COLUMN.TYPE}
{/TABLE.COLUMNS}
)   
AS  
 DELETE FROM {TABLE.NAME}   
 WHERE {TABLE.COLUMNS PRIMARY}{COLUMN.NAME} = @{COLUMN.NAME}{/TABLE.COLUMNS}
GO


CREATE PROCEDURE sp{TABLE.NAME}_List  
AS  
SELECT {TABLE.COLUMNS}{COLUMN.NAME}{IF NOT LAST},{/IF}{/TABLE.COLUMNS}
FROM {TABLE.NAME}   

GO
